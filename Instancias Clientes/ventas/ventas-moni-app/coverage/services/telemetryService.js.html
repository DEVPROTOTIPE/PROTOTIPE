/**
 * Genera un hash/firma simple para identificar errores idénticos.
 */
function getErrorHash(errorMsg, stack) {
  const cleanStack = (stack || '').split('\n')[0] || '';
  return `${errorMsg}_${cleanStack}`.replace(/[^a-zA-Z0-9_]/g, '_');
}
 
/**
 * Genera un eventId único para idempotencia de reportes.
 */
function generateEventId(type, suffix = '') {
  return `${type}-${CLIENT_ID || 'anon'}-${Date.now()}-${suffix || Math.random().toString(36).slice(2)}`;
}
 
/**
 * Encola un reporte offline en IndexedDB (Dexie).
 * Reemplaza la implementación anterior basada en localStorage.
 * La cola es transaccional y no tiene límite de tamaño efectivo.
 */
async function enqueueOfflineReport(type, payload) {
  try {
    const eventId = payload.eventId || generateEventId(type);
    await telemetryDb.outbox.put({
      eventId,
      type,
      payload: { ...payload, eventId },
      status: 'pending',
      attempts: 0,
      createdAt: Date.now(),
      nextAttemptAt: Date.now(),
    });
    console.debug(`[Telemetry] Reporte '${type}' encolado en IndexedDB (eventId: ${eventId}).`);
  } catch (err) {
    console.error('[Telemetry] Error al encolar reporte en IndexedDB:', err);
  }
}

            lastError: String(err?.message ?? err),

        await telemetryDb.outbox.delete(item.eventId);
      } catch (err) {
        const attempts = (item.attempts || 0) + 1;
 
        if (attempts >= MAX_ATTEMPTS) {
          await telemetryDb.outbox.update(item.eventId, {
            status: 'failed',
            attempts,
            lastError: String(err?.message ?? err),
          });
          console.warn(`[Telemetry] Reporte billing '${item.eventId}' marcado como fallido tras ${MAX_ATTEMPTS} intentos.`);

          lastError: String(err?.message ?? err),
        });

      }
    }

/**
 * Lógica de envío del reporte mensual de facturación (Billing)
 */
async function executeBillingReport(payload) {
  console.log("[Telemetry] Reportando facturación mensual...");
  
  const centralDb = getCentralFirestore();
  if (!centralDb) {
    throw new Error("No hay conexión a la base de datos central.");
  }
 
  // Curar reportes antiguos encolados en la cola local offline (legacy sin token)
  if (!payload.token && DEV_TOKEN) {
    payload.token = DEV_TOKEN;
  }
 
  console.log("[Telemetry] Escribiendo reporte directo en Firestore Central (reportesBilling)...");
  const reportId = `${payload.clientId}_${payload.periodo}`;
  const docRef = doc(centralDb, 'reportesBilling', reportId);
  await setDoc(docRef, {
    ...payload,
    updatedAt: serverTimestamp()
  });

}

const NOISE_TO_IGNORE = [
  'failed to fetch',
  'load failed',
  'networkerror',
  'script error',
  'extension',
  'cors',
  'canceled'

/**
 * Reporta un error o excepción de la aplicación a la base de datos central de errores.
 * Integra rate-limiting local (Circuit Breaker) para evitar DDoS accidentales desde el cliente.
 */
export async function reportAppFailureToDeveloper(errorMsg, stack, source = 'automatic') {
  if (!DEV_TOKEN || !CLIENT_ID) return;
 
  const nowTime = Date.now();
 
  // Resetear contador de tasa cada minuto
  if (nowTime - minuteResetTimestamp >= 60000) {
    reportsThisMinute = 0;
    minuteResetTimestamp = nowTime;
  }
 
  // Verificar si el interruptor está activo (Cooldown de 5 minutos)
  if (circuitBreakerTripped) {
    if (nowTime < circuitBreakerResetAt) {
      console.warn(`[Telemetry] Circuit Breaker activo. Omitiendo reporte de error para evitar sobrecostos.`);
      return;
    } else {
      circuitBreakerTripped = false;
      console.log(`[Telemetry] Cooldown de Circuit Breaker finalizado. Telemetría reactivada.`);
    }
  }
 
  const msgLower = (errorMsg || '').toLowerCase();
 
  // 1. Filtrar en frío errores automáticos de red/extensiones no críticos
  if (source === 'automatic') {
    const isNoise = NOISE_TO_IGNORE.some(patron => msgLower.includes(patron));
    if (isNoise) {
      console.debug(`[Telemetry] Filtro de ruido activo. Omitiendo reporte automático: ${errorMsg}`);
      return;
    }
  }
 
  // 2. Mecanismo Anti-Duplicado (Throttle de 5 minutos / 300 segundos por firma de error)
  const errorHash = getErrorHash(errorMsg, stack);
  if (reportedErrorsCache[errorHash] && (nowTime - reportedErrorsCache[errorHash] < 300000)) {
    console.debug(`[Telemetry] Reporte de error duplicado omitido (Throttled): ${errorMsg}`);
    return;
  }
  reportedErrorsCache[errorHash] = nowTime;
 
  // 3. Comprobar límite de tasa de telemetría (Circuit Breaker)
  if (reportsThisMinute >= MAX_REPORTS_PER_MINUTE) {
    circuitBreakerTripped = true;
    circuitBreakerResetAt = nowTime + 300000; // 5 minutos de cooldown
    console.error(`[Telemetry] Límite de ${MAX_REPORTS_PER_MINUTE} reportes/minuto alcanzado. Circuit Breaker saltado (cooldown de 5 minutos).`);
    
    // Registrar una única alerta en IndexedDB sobre la desconexión
    await enqueueOfflineReport('failure', {
      clientId: CLIENT_ID,
      token: DEV_TOKEN || 'desconocido',
      niche: CLIENT_NICHE,
      timestamp: new Date().toISOString(),
      errorMsg: `Circuit Breaker Tripped: Excedido el límite de ${MAX_REPORTS_PER_MINUTE} reportes/minuto`,
      stack: 'Circuit breaker local activado por rate-limiting de telemetría.',
      source: 'telemetry_system',
      environment: {
        url: typeof window !== 'undefined' ? window.location.href : 'N/A',
        userAgent: typeof navigator !== 'undefined' ? navigator.userAgent : 'Server/Node'
      }
    });
    return;
  }
 
  reportsThisMinute++;
 
  // Contexto del usuario logueado en Firebase Auth (Seguro sin datos sensibles)
  let userContext = null;
  if (auth && auth.currentUser) {
    userContext = {
      uid: auth.currentUser.uid,
      email: auth.currentUser.email
    };
  }
 
  const extendedPayload = {
    clientId: CLIENT_ID,
    token: DEV_TOKEN || "desconocido",
    niche: CLIENT_NICHE,
    timestamp: new Date().toISOString(),
    errorMsg: errorMsg || 'Unknown Error',
    stack: stack || 'No stack trace available',
    resolved: false,
    source,
    environment: {
      url: typeof window !== 'undefined' ? window.location.href : 'N/A',
      userAgent: typeof navigator !== 'undefined' ? navigator.userAgent : 'Server/Node',
      language: typeof navigator !== 'undefined' ? navigator.language : 'N/A',
      screenResolution: typeof window !== 'undefined' ? `${window.screen.width}x${window.screen.height}` : 'N/A',
      viewport: typeof window !== 'undefined' ? `${window.innerWidth}x${window.innerHeight}` : 'N/A'
    },
    user: userContext
  };
 
  // Siempre encolamos localmente en IndexedDB para loteo diferido y resiliencia offline
  await enqueueOfflineReport('failure', extendedPayload);
  processOfflineQueue().catch(e => console.debug('[Telemetry] Flush error:', e.message));
}
